Thank you for installing {{ .Chart.Name }}!

Your Travelo application has been deployed to namespace: {{ .Values.global.namespace }}

To get the application URL:
{{- if .Values.proxy.enabled }}
  {{- if eq .Values.proxy.service.type "LoadBalancer" }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Values.global.namespace }} travelo-proxy --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Visit http://$SERVICE_IP to access your application"
  {{- else if eq .Values.proxy.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Values.global.namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services travelo-proxy)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Values.global.namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Visit http://$NODE_IP:$NODE_PORT to access your application"
  {{- else }}
  kubectl port-forward --namespace {{ .Values.global.namespace }} svc/travelo-proxy 8080:80
  echo "Visit http://localhost:8080 to access your application"
  {{- end }}
{{- end }}

To check the status of your deployment:
  kubectl get pods -n {{ .Values.global.namespace }}
  kubectl get services -n {{ .Values.global.namespace }}

To view logs:
  kubectl logs -n {{ .Values.global.namespace }} -l app.kubernetes.io/name=travelo -l app.kubernetes.io/component=backend
  kubectl logs -n {{ .Values.global.namespace }} -l app.kubernetes.io/name=travelo -l app.kubernetes.io/component=frontend
